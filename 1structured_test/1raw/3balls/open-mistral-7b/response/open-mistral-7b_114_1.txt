C(n + k-1, k-1, k-1, ..., k-1)

C(6 + 3-1, 3-1, 3-1, 3-1) = C(8, 2, 2, 2)

C(6, 2) * C(4, 2) * C(2, 2)

C(6, 2) * C(4, 3) * C(2, 1)
