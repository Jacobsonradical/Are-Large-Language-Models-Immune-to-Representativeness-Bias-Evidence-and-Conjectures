\[ P(B \mid E) = \frac{P(E \mid B) \cdot P(B)}{P(E)} \]

\[ P(E) = P(E \mid B) \cdot P(B) + P(E \mid W) \cdot P(W) \]
\[ P(E) = 0.8 \cdot 0.15 + 0.2 \cdot 0.85 \]
\[ P(E) = 0.12 + 0.17 \]

\[ P(B \mid E) = \frac{0.8 \cdot 0.15}{0.29} \]
\[ P(B \mid E) = \frac{0.12}{0.29} \]
