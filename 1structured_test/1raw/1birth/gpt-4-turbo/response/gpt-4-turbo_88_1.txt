\[ P(\text{G B G B B G}) = \left(\frac{1}{2}\right)^6 = \frac{1}{64} \]

\[ \frac{45}{\frac{1}{64}} = 45 \times 64 = 2880 \]

\[ P(\text{B G B B B B}) = \left(\frac{1}{2}\right)^6 = \frac{1}{64} \]

\[ \text{Estimated number of families with B G B B B B} = 2880 \times \frac{1}{64} = 45 \]
