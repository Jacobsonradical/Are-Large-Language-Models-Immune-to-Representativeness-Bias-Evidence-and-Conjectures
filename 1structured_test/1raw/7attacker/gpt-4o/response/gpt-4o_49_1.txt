\[ 
P(T|W) = \frac{P(W|T) \cdot P(T)}{P(W)} 
\]

P(W) = P(W|T) \cdot P(T) + P(W|\neg T) \cdot P(\neg T)

P(W) = (0.80 \times 0.15) + (0.20 \times 0.85) = 0.12 + 0.17 = 0.29 
\]

\[ 
P(T|W) = \frac{P(W|T) \cdot P(T)}{P(W)} 
\]
\[ 
P(T|W) = \frac{0.80 \times 0.15}{0.29} = \frac{0.12}{0.29} \approx 0.4138
