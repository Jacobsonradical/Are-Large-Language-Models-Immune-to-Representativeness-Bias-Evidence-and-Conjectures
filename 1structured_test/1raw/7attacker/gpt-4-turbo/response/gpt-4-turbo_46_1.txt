\[ P(T|W) = \frac{P(W|T) \times P(T)}{P(W)} \]

\[ P(W) = P(W|T) \times P(T) + P(W|S) \times P(S) \]

\[ P(T|W) = \frac{(0.8 \times 0.15)}{0.29} = \frac{0.12}{0.29} \approx 0.4138 \]
