P(B) = P(B|A) * P(A) + P(B|A') * P(A')

P(A) = (P(B) - P(B|A') * P(A')) / P(B|A)

P(B) = P(B and A) + P(B and A')
P(B) = P(B|A) * P(A) + P(B|A') * P(A')

P(B) = (0.8 * P(A)) + (0.2 * 0.85)

P(A) = ([0.8 * P(A)] + (0.2 * 0.85) - 0.2 * 0.85) / 0.8
