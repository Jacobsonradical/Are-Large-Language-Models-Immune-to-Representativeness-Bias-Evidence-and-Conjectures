\[ P(\text{Father is Nondreamer} | \text{Brett is Nondreamer}) = \frac{P(\text{Brett is Nondreamer} | \text{Father is Nondreamer}) \cdot P(\text{Father is Nondreamer})}{P(\text{Brett is Nondreamer})} \]

\[ P(\text{Brett is Nondreamer} | \text{Father is Nondreamer}) = P(\text{Brett is Nondreamer}) = 0.20 \]

\[ P(\text{Father is Nondreamer} | \text{Brett is Nondreamer}) = \frac{0.20 \cdot 0.20}{0.20} = 0.20 \]
