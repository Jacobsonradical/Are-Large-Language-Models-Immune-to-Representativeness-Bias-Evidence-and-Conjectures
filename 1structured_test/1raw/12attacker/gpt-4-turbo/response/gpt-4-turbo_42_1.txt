\[ P(A|B) = \frac{P(B|A) \times P(A)}{P(B)} \]

\[ P(B) = P(B|A) \times P(A) + P(B|A^c) \times P(A^c) \]
\[ P(B) = (0.8 \times 0.15) + (0.2 \times 0.85) \]
\[ P(B) = 0.12 + 0.17 = 0.29 \]

\[ P(A|B) = \frac{0.8 \times 0.15}{0.29} = \frac{0.12}{0.29} \approx 0.4138 \]
