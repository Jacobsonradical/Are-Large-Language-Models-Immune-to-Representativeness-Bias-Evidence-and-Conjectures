\[
P(M | W_m) = \frac{P(W_m | M) \cdot P(M)}{P(W_m)}

\[
P(W_m) = P(W_m | M) \cdot P(M) + P(W_m | F) \cdot P(F)

\[
P(W_m) = 0.8 \cdot 0.15 + 0.2 \cdot 0.85

\[
