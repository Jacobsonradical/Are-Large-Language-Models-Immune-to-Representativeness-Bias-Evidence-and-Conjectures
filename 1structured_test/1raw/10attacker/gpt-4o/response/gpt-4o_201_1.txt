\[ P(H_1 | W_1, W_2) = \frac{P(W_1, W_2 | H_1) \cdot P(H_1)}{P(W_1, W_2)} \]

\[ P(W_1, W_2) = P(W_1, W_2 | H_1) \cdot P(H_1) + P(W_1, W_2 | H_2) \cdot P(H_2) \]

\[ P(H_1 | W_1, W_2) = \frac{0.036}{0.155} \]
