\[ P(W1 = H, W2 = \neg H \mid H) = P(W1 = H \mid H) \cdot P(W2 = \neg H \mid H) = 0.8 \cdot 0.3 \]
\[ P(W1 = H, W2 = \neg H \mid \neg H) = P(W1 = H \mid \neg H) \cdot P(W2 = \neg H \mid \neg H) = 0.2 \cdot 0.7 \]

\[ P(W1 = H, W2 = \neg H) = P(H) \cdot P(W1 = H, W2 = \neg H \mid H) + P(\neg H) \cdot P(W1 = H, W2 = \neg H \mid \neg H) \]
\[ P(W1 = H, W2 = \neg H) = 0.15 \cdot (0.8 \cdot 0.3) + 0.85 \cdot (0.2 \cdot 0.7) \]

\[ P(W1 = H, W2 = \neg H) = 0.036 + 0.119 \]
\[ P(W1 = H, W2 = \neg H) = 0.155 \]

\[ P(H \mid W1 = H, W2 = \neg H) = \frac{P(H) \cdot P(W1 = H, W2 = \neg H \mid H)}{P(W1 = H, W2 = \neg H)} \]

\[ P(H \mid W1 = H, W2 = \neg H) = \frac{0.036}{0.155} \]
