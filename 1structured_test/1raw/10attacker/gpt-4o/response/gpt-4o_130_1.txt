\[
P(H | W1) = \frac{P(W1 | H) \cdot P(H)}{P(W1)}

\[
P(W1) = P(W1 | H) \cdot P(H) + P(W1 | L) \cdot P(L)

\[

\[
P(H | W1) = \frac{0.80 \cdot 0.15}{0.29} = \frac{0.12}{0.29} \approx 0.4138
