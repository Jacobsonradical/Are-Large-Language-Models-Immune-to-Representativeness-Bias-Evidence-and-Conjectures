\[
P(H \mid W_1) = \frac{P(W_1 \mid H) \times P(H)}{P(W_1)}
\]

\[
P(W_1) = P(W_1 \mid H) \times P(H) + P(W_1 \mid \neg H) \times P(\neg H)

\[

\[
