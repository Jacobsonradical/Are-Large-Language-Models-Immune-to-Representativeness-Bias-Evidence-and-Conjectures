\[
P(A | W_1) = \frac{P(W_1 | A) P(A)}{P(W_1)}

\[

\]
\[
P(A | W_1) = \frac{0.12}{0.29} \approx 0.4138
\]

\[
P(A | W_1, W_2) = \frac{P(W_2 | A) P(A | W_1)}{P(W_2 | W_1)}

\[
