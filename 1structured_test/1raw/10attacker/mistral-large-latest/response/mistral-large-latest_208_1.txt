\[ P(W1, W2|T) = P(W1|T) \times P(W2|T) = 0.80 \times 0.30 = 0.24 \]

\[ P(W1, W2) = P(W1, W2|T) \times P(T) + P(W1, W2|\neg T) \times P(\neg T) \]

\[ P(T|W1, W2) = \frac{P(W1, W2|T) \times P(T)}{P(W1, W2)} \]

\[ P(T|W1, W2) = \frac{0.036}{0.155} \approx 0.232 \]
