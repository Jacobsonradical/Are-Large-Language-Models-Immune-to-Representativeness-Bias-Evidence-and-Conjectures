\[
P(T | W) = \frac{P(W | T) P(T)}{P(W)}
\]

P(T) = 0.5 \text{ (since the population is equally divided)}

\[
P(\neg T) = 0.5

P(W | T) = 0.8 \text{ (the witness correctly identifies the height 80% of the time)}

\[
P(W) = P(W | T) P(T) + P(W | \neg T) P(\neg T)

\[
P(W) = 0.8 \times 0.5 + 0.2 \times 0.5

P(W) = 0.4 + 0.1 = 0.5

P(T | W) = \frac{P(W | T) P(T)}{P(W)}
