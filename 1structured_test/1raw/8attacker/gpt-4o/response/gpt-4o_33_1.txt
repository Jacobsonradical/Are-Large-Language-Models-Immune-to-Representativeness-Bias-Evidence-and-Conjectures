\[
P(W) = P(W|T)P(T) + P(W|S)P(S)

P(W) = (0.8 \times 0.5) + (0.2 \times 0.5)

\[
P(T|W) = \frac{P(W|T)P(T)}{P(W)}
\]
\[
P(T|W) = \frac{0.8 \times 0.5}{0.5}
\]
\[
P(T|W) = \frac{0.4}{0.5} = 0.8
