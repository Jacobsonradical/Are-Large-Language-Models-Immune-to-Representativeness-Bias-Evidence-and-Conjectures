\[
P(T|W) = \frac{P(W|T) \cdot P(T)}{P(W)}
\]

\[
P(W) = P(W|T) \cdot P(T) + P(W|T') \cdot P(T')

\[
P(W) = (0.80 \cdot 0.50) + (0.20 \cdot 0.50) = 0.40 + 0.10 = 0.50

\[
