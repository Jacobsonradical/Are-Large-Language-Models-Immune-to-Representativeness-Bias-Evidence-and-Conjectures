\[ P(A|W) = \frac{P(W|A)P(A)}{P(W)} \]

\[ P(W) = P(W|A)P(A) + P(W|\neg A)P(\neg A) \]
\[ P(W) = (0.8 \times 0.15) + (0.2 \times 0.85) \]
\[ P(W) = 0.12 + 0.17 = 0.29 \]

\[ P(A|W) = \frac{0.8 \times 0.15}{0.29} \]
\[ P(A|W) = \frac{0.12}{0.29} \approx 0.4138 \]
