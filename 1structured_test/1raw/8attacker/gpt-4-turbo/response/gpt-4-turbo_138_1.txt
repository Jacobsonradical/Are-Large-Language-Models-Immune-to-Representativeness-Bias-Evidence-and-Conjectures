   \[ P(A|W) = \frac{P(W|A)P(A)}{P(W)} \]
   \[ P(A|W) = \frac{(0.80)(0.15)}{0.29} \]
   \[ P(A|W) = \frac{0.12}{0.29} \approx 0.4138 \]
