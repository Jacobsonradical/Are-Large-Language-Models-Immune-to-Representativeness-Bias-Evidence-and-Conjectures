\[ P(H|E) = \frac{P(E|H) \times P(H)}{P(E)} \]

\[ P(E) = P(E|H) \times P(H) + P(E|~H) \times P(~H) \]
\[ P(E) = (0.80 \times 0.15) + (0.20 \times 0.85) \]
\[ P(E) = 0.12 + 0.17 = 0.29 \]

\[ P(H|E) = \frac{0.80 \times 0.15}{0.29} \]
\[ P(H|E) = \frac{0.12}{0.29} \approx 0.4138 \]
